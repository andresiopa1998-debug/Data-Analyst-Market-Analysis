SELECT 
    experience_level,
    ROUND(AVG(salary), 2) AS avg_salary,
    COUNT(*) AS job_count
FROM 
    `coursera-projects-487321.job_market_analysis.AI_Job_MarketDataset`
WHERE 
    job_title = 'Data Analyst'
GROUP BY 
    experience_level
ORDER BY 
    avg_salary DESC;
